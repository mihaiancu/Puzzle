\section{Function shift is injective}

\bigskip

\begin{lemma}
\label{fun_shift_surj}
\lean{fun_shift_surj}
\leanok
\uses{def:SF}
If $f:\Zz^n \to \Fin n$ is \eqref{SF} and $b\in\Zz^n$,
then the function $\Fin n \ni i \mapsto f(b+{\rm e}_i) \in \Fin n$ is {\B surjective}.
\end{lemma}

\begin{proof}
$\Fin n \ni i \mapsto f(b+{\rm e}_i) \in \Fin n$ is surjective, if
$$\forall k \in \Fin n, \exists i \in \Fin n, f(b+{\rm e}_i) = k.$$
Taking our $b$ in \eqref{SF}, we get the desired conclusion.
\end{proof}

\bigskip

\begin{lemma}
\label{fun_shift_inj}
\lean{fun_shift_inj}
\leanok
\uses{def:SF}
If $f:\Zz^n \to \Fin n$ is \eqref{SF} and $b\in\Zz^n$,
    then the function $\Fin n \ni i \mapsto f(b+{\rm e}_i) \in \Fin n$ is {\B injective}.
\end{lemma}

\begin{proof}
\uses{fun_shift_surj}
Since $\Fin n \ni i \mapsto f(b+{\rm e}_i) \in \Fin n$ is a {\B surjective} function, by {\B Lemma} \ref{fun_shift_surj}, on a {\B finite} set, it is also {\B injective}.
\end{proof}
